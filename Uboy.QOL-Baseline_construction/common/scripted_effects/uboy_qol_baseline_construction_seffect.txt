# ========================================
# Scripted Effect: 편입주에 건물 건설
# ========================================
uboy_qol_baseline_construction_effect_build_incorporated = {
    every_scope_state = {
        limit = {
            is_incorporated = yes
            owner = ROOT
            NOT = { has_building = $BUILDING$ }
        }
        start_building_construction = $BUILDING$
    }
}

# ========================================
# Scripted Effect: 비편입주에 건물 건설
